3. 导航系统

macOS 风格 Dock 导航

采用 macOS 底部 Dock 风格导航，悬浮自动隐藏 + Pill 指示器标记当前页面，提供沉浸式数据浏览体验。

设计原则

  • 沉浸式体验：导航栏悬浮于内容之上，不占用布局空间
  • 视觉反馈：Pill 指示器持久化显示当前位置，鼠标悬停放大效果
  • 自动隐藏：鼠标离开底部区域后自动隐藏，最大化数据展示区域
  • 平滑动画：所有状态切换均使用 CSS transition，过渡自然

5. Composables 模式

5.1 useECharts

统一管理 ECharts 实例的生命周期，自动处理 resize 和销毁。

// composables/useECharts.ts
export function useECharts(chartRef: Ref<HTMLElement | null>) {
  let chartInstance: ECharts | null = null

  const initChart = () => {
    if (!chartRef.value) return
    chartInstance = echarts.init(chartRef.value, 'dark')
  }

  const setOption = (option: EChartsOption) => {
    chartInstance?.setOption(option, { notMerge: true })
  }

  const resize = () => chartInstance?.resize()

  onMounted(() => {
    initChart()
    window.addEventListener('resize', resize)
  })

  onUnmounted(() => {
    window.removeEventListener('resize', resize)
    chartInstance?.dispose()
    chartInstance = null
  })

  return { setOption, resize }
}

5.2 useFundSearch

封装基金搜索逻辑，支持防抖和结果缓存。

// composables/useFundSearch.ts
export function useFundSearch() {
  const query = ref('')
  const results = ref<Fund[]>([])
  const loading = ref(false)

  const search = useDebounceFn(async () => {
    if (!query.value.trim()) {
      results.value = []
      return
    }
    loading.value = true
    try {
      results.value = await fundApi.search(query.value)
    } finally {
      loading.value = false
    }
  }, 300)

  return { query, results, loading, search }
}

5.3 useTheme

管理主题切换，持久化用户偏好。

// composables/useTheme.ts
export function useTheme() {
  const isDark = useLocalStorage('theme-dark', true)

  const toggle = () => {
    isDark.value = !isDark.value
    document.documentElement.classList.toggle('dark', isDark.value)
  }

  onMounted(() => {
    document.documentElement.classList.toggle('dark', isDark.value)
  })

  return { isDark, toggle }
}

6. API 层设计

6.1 Axios 实例与拦截器

// api/client.ts
const client = axios.create({
  baseURL: import.meta.env.VITE_API_BASE,
  timeout: 10000,
})

// 请求拦截器：自动取消重复 GET
const pendingRequests = new Map<string, AbortController>()

client.interceptors.request.use((config) => {
  if (config.method === 'get') {
    const key = config.url + JSON.stringify(config.params)
    const existing = pendingRequests.get(key)
    if (existing) existing.abort()
    const controller = new AbortController()
    config.signal = controller.signal
    pendingRequests.set(key, controller)
  }
  return config
})

// 响应拦截器：统一解包 + CancelError 处理
client.interceptors.response.use(
  (res) => {
    const key = res.config.url! + JSON.stringify(res.config.params)
    pendingRequests.delete(key)
    return res.data  // 统一解包
  },
  (err) => {
    if (axios.isCancel(err)) {
      return Promise.reject(new CancelError('请求已取消'))
    }
    return Promise.reject(err)
  }
)

7. Store 层设计

7.1 FundStore

// stores/fund.ts
export const useFundStore = defineStore('fund', () => {
  const watchlist = ref<Fund[]>([])
  const currentFund = ref<FundDetail | null>(null)
  const trendData = ref<TrendPoint[]>([])

  const fetchDetail = async (code: string) => {
    currentFund.value = await fundApi.getDetail(code)
  }

  const fetchTrend = async (code: string) => {
    trendData.value = await fundApi.getTrend(code)
  }

  return { watchlist, currentFund, trendData, fetchDetail, fetchTrend }
})

7.2 PredictionStore

// stores/prediction.ts
export const usePredictionStore = defineStore('prediction', () => {
  const prediction = ref<PredictionResult | null>(null)
  const features = ref<FeatureImportance[]>([])
  const backtestData = ref<BacktestPoint[]>([])

  const fetchPrediction = async (code: string) => {
    prediction.value = await predictionApi.predict(code)
  }

  const fetchFeatures = async (code: string) => {
    features.value = await predictionApi.getFeatures(code)
  }

  const fetchBacktest = async (code: string) => {
    backtestData.value = await predictionApi.getBacktest(code)
  }

  return { prediction, features, backtestData, fetchPrediction, fetchFeatures, fetchBacktest }
})
